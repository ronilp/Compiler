main()
{
    protein P[20]
    P is "AAGTCCAGGATGCA"
    int i
    repeat i from 0 to length(P)
    {
        condition(P[i]="T")
        {
            write("DNA")
            end
        }
        condition(P[i]="U")
        {
            write("RNA")
            end
        }
    }
    write("Neither DNA nor RNA")
    return
}
