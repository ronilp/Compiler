function void left(protein P[],int index)
{
    int i
    write("left part ")
    repeat i from 0 to index
    {
        write(P[i])
    }
    write(" ")
    return
}
function void right(protein P[],int index)
{
    int i
    write("right part ")
    repeat i from index+3 to length(P)
    {
        write(P[i])
    }
    write(" ")
    return
}
